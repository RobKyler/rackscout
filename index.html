// Unique key (text, preferred — auto-parsed from chat_link)
thread_id: string — digits from messenger.com/marketplace/t/<thread_id>/

// Primary key (bigint — auto-derived from thread_id if present)
id: number — optional if thread_id or chat_link provided

// Required fields
name: string — buyer full name
status: "pending" | "scheduled" | "sold" | "cancelled"

// Optional fields
source: string — default "Facebook Marketplace"
item: string — default "Storage Racks"
price: string — e.g. "$75" or "$60" (per unit, not total)
units: number — quantity buyer wants
meetup: string — e.g. "TBD — buyer picking up"
datetime: string — ISO "2026-07-05T12:00" or omit
notes: string — conversation details
suggested_reply: string — reply script
chat_link: string — full messenger URL (thread_id auto-extracted)
added_at: string — e.g. "6/28/2026, 12:00:00 PM"
